def before
  if env['PATH_INFO'].index(prefix) == 0
    env['PATH_INFO'].sub!(prefix, '')
    env['PATH_INFO'] = Rack::Mount::Utils.normalize_path(env['PATH_INFO'])
  end
end

def prefix

  prefix
