def to_app(app)
  b = Rack::Builder.new
  for middleware in stack
    b.use *middleware
  end
  b.run app
  b.to_app
end

def use(klass, *args)
  if index = @stack.index(@stack.find{|a| a.first == klass})
    @stack[index] = [klass, *args]
  else
    @stack << [klass, *args]
  end
end
